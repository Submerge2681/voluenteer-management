CREATE OR REPLACE VIEW checkin_list AS
SELECT
  profiles.id AS user_id,
  profiles.full_name,
  participation.event_id
FROM participation
JOIN profiles ON profiles.id = participation.user_id;



create or replace function get_top_cities()
returns table (city text, participants bigint) 
language sql
security definer
as $$
  select 
    trim(location) as city, 
    -- Sum participants (handle nulls with coalesce)
    sum(coalesce(participants, 0)) as participants
  from events
  where is_completed = true
  group by 1
  order by 2 desc
  limit 5;
$$;


create or replace function recalculate_global_stats()
returns void
language plpgsql
security definer
as $$
declare
  _total_volunteers int;
  _total_waste int;
  _total_cleanups int;
begin
  -- Calculate Total Volunteers (Sum of participants in completed events)
  select coalesce(sum(participants), 0) 
  into _total_volunteers 
  from events 
  where is_completed = true;

  -- Calculate Total Waste
  select coalesce(sum(waste_kg), 0) 
  into _total_waste 
  from events 
  where is_completed = true;

  -- Calculate Total Cleanups
  select count(*) 
  into _total_cleanups 
  from events 
  where is_completed = true;

  -- Update the stats table (Assuming ID 1 is your main stats row)
  update global_stats
  set 
    total_volunteers = _total_volunteers,
    total_waste = _total_waste,
    total_cleanups = _total_cleanups,
    last_updated = now()
  where id = 1;

end;
$$;


select cron.schedule(
  'update-daily-stats', -- name of the cron job
  '0 0 * * *',          -- cron syntax (every day at midnight)
  $$select recalculate_global_stats()$$
);


-- ============================================================
-- Rate Limits Table
-- ============================================================
CREATE TABLE IF NOT EXISTS rate_limits (
  ip             text        NOT NULL,
  window_start   timestamptz NOT NULL,
  count          integer     NOT NULL DEFAULT 0,
  PRIMARY KEY (ip, window_start)
);

-- Index to make cleanup fast
CREATE INDEX IF NOT EXISTS rate_limits_window_start_idx
  ON rate_limits (window_start);

-- Only accessible via the SECURITY DEFINER function below
ALTER TABLE rate_limits ENABLE ROW LEVEL SECURITY;
-- No RLS policies — the function's SECURITY DEFINER handles access

-- ============================================================
-- Atomic Rate Limit Check (fixed window)
-- ============================================================
CREATE OR REPLACE FUNCTION check_rate_limit(
  p_ip             text,
  p_max_requests   integer DEFAULT 10,
  p_window_seconds integer DEFAULT 60
)
RETURNS boolean
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_window  timestamptz;
  v_count   integer;
BEGIN
  -- Align to fixed window boundary
  v_window := to_timestamp(
    floor(extract(epoch FROM now()) / p_window_seconds) * p_window_seconds
  );

  INSERT INTO rate_limits (ip, window_start, count)
  VALUES (p_ip, v_window, 1)
  ON CONFLICT (ip, window_start)
  DO UPDATE SET count = rate_limits.count + 1
  RETURNING count INTO v_count;

  RETURN v_count <= p_max_requests;
END;
$$;

-- Grant execute to the anon/authenticated roles used by your supabase client
GRANT EXECUTE ON FUNCTION check_rate_limit(text, integer, integer)
  TO anon, authenticated;

-- ============================================================
-- Cleanup Function (call via pg_cron or manually)
-- ============================================================
CREATE OR REPLACE FUNCTION cleanup_rate_limits(
  older_than interval DEFAULT '1 year'
)
RETURNS void
LANGUAGE sql
SECURITY DEFINER
SET search_path = public
AS $$
  DELETE FROM rate_limits
  WHERE window_start < now() - older_than;
$$;

-- ============================================================
-- Schedule cleanup with pg_cron (enable via Supabase dashboard)
-- ============================================================
SELECT cron.schedule(
  'cleanup-rate-limits',
  '0 0 4 6- *',
  'SELECT cleanup_rate_limits();'
);



create or replace function sync_event_participants()
returns trigger as $$
declare
  target_event_id uuid;
begin
  if TG_OP = 'DELETE' then
    target_event_id := OLD.event_id;
  else
    target_event_id := NEW.event_id;
  end if;

  update events
  set participants = (
    select count(*) from participation where event_id = target_event_id
  )
  where id = target_event_id;

  return NEW;
end;
$$ language plpgsql;

create trigger trg_sync_participants
after insert or delete on participation
for each row execute function sync_event_participants();
