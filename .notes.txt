  1. Critical Bug: Guest TOTP Flow Will Almost Always Fail                                                                                                                                                                                                                                                                  
                  
  File: src/app/auth/callback/route.ts:39 and src/app/api/checkin/route.ts:65

  The guest flow stores the raw TOTP code in the cookie, then re-validates it up to 15 minutes later in the callback. A TOTP code expires every 30 seconds. By the time the user signs up and clicks the magic link, the code is long expired, validation silently fails, and they're redirected to the dashboard without
  being checked in.

  Fix: Don't store the raw OTP. After successful validation, store a pre-validated claim instead:

  // In route.ts — after validation succeeds, set:
  response.cookies.set('pending_checkin', JSON.stringify({
    validated: true,
    event_id
  }), { ... });

  // In callback/route.ts — skip re-validation:
  const { validated, event_id } = JSON.parse(pendingCookie.value);
  if (validated && event_id && user) {
    await supabase.from('participation').insert({ user_id: user.id, event_id });
  }

  You should also HMAC-sign the cookie value to prevent forgery (since you skip re-validation). Use a COOKIE_SECRET env var with Node's crypto.createHmac.

  2. Critical Bug: otplib API Used Incorrectly — TOTP Likely Never Validates

  File: src/lib/totp.ts:39-46

  // This is wrong:
  const result = await verify({ secret, token: trimmedCode });
  if (result.valid) { ... }

  otplib's verify returns a synchronous boolean, not a promise with a .valid property. So result is true or false, and result.valid is always undefined (falsy). Every valid TOTP code returns { success: false }. Same problem with generate({ secret }) — it expects generate(secret) as a positional string.

  Fix: Use the authenticator preset directly:

  import { authenticator } from 'otplib';

  export async function validateEventCheckin(...) {
    if (event.checkin_type === 'totp') {
      const isValid = authenticator.verify({
        token: trimmedCode,
        secret: event.checkin_secret
      });
      return isValid
        ? { success: true, message: 'TOTP Verified' }
        : { success: false, message: 'Code expired or invalid' };
    }
  }

  export function generateEventSecret(type: 'totp' | 'static_otp') {
    if (type === 'totp') return authenticator.generateSecret();
    // ...
  }

  export async function generateEventCode(secret: string, type: 'totp' | 'static_otp') {
    if (type === 'static_otp') return secret;
    return authenticator.generate(secret);
  }

  3. Security: TOTP Secret Sent to the Browser

  File: src/app/admin/scan/[id]/page.tsx:55, src/app/admin/scan/[id]/AdminCheckinManager.tsx:7

  The raw checkin_secret is passed as a prop to a 'use client' component, which means it's serialized into the page HTML and sent to the browser. Any admin who opens DevTools can extract it and generate unlimited valid TOTP codes.

  Fix: Move code generation server-side. Create a Server Action that generates the current code and remaining time, and have the client component poll it every second instead of receiving the secret:

  // New server action
  'use server'
  export async function getCurrentCode(eventId: string) {
    // auth check here
    const { data: event } = await supabase.from('events')
      .select('checkin_secret, checkin_type').eq('id', eventId).single();
    return { code: await generateEventCode(event.checkin_secret, event.checkin_type), timeLeft: getTotpTimeRemaining() };
  }

  // AdminCheckinManager.tsx — poll instead of holding secret
  const { eventId, type } = props; // no `secret` prop
  useEffect(() => {
    const tick = async () => {
      const { code, timeLeft } = await getCurrentCode(eventId);
      setCode(code); setTimeLeft(timeLeft);
    };
    tick();
    const timer = setInterval(tick, 1000);
    return () => clearInterval(timer);
  }, [eventId]);

  4. Security: OTP Exposed in GET Request URL

  File: src/app/admin/scan/[id]/AdminCheckinManager.tsx:45

  const url = `${origin}/api/checkin?otp=${code}&event_id=${eventId}`;

  The OTP appears in browser history, server logs, and Referer headers. You acknowledge this in a comment but keep the GET route anyway. Since the QR code is scanned with a phone camera which opens a browser, this is unavoidable for the QR use-case. However, you can mitigate it:

  - Make the GET route redirect to a landing page that completes check-in via POST, keeping the OTP out of logs after the redirect.
  - Or generate a short-lived one-time token server-side and encode that in the URL instead of the raw OTP.

  if (validation.success) {

create or replace function get_top_cities()
returns table (city text, participants bigint) 
language sql
security definer
as $$
  select 
    trim(location) as city, 
    -- Sum participants (handle nulls with coalesce)
    sum(coalesce(participants, 0)) as participants
  from events
  where is_completed = true
  group by 1
  order by 2 desc
  limit 5;
$$;


create or replace function recalculate_global_stats()
returns void
language plpgsql
security definer
as $$
declare
  _total_volunteers int;
  _total_waste int;
  _total_cleanups int;
begin
  -- Calculate Total Volunteers (Sum of participants in completed events)
  select coalesce(sum(participants), 0) 
  into _total_volunteers 
  from events 
  where is_completed = true;

  -- Calculate Total Waste
  select coalesce(sum(waste_kg), 0) 
  into _total_waste 
  from events 
  where is_completed = true;

  -- Calculate Total Cleanups
  select count(*) 
  into _total_cleanups 
  from events 
  where is_completed = true;

  -- Update the stats table (Assuming ID 1 is your main stats row)
  update global_stats
  set 
    total_volunteers = _total_volunteers,
    total_waste = _total_waste,
    total_cleanups = _total_cleanups,
    last_updated = now()
  where id = 1;

end;
$$;

-- ============================================================
-- Rate Limits Table
-- ============================================================
CREATE TABLE IF NOT EXISTS rate_limits (
  ip             text        NOT NULL,
  window_start   timestamptz NOT NULL,
  count          integer     NOT NULL DEFAULT 0,
  PRIMARY KEY (ip, window_start)
);

-- Index to make cleanup fast
CREATE INDEX IF NOT EXISTS rate_limits_window_start_idx
  ON rate_limits (window_start);

-- Only accessible via the SECURITY DEFINER function below
ALTER TABLE rate_limits ENABLE ROW LEVEL SECURITY;
-- No RLS policies — the function's SECURITY DEFINER handles access

-- ============================================================
-- Atomic Rate Limit Check (fixed window)
-- ============================================================
CREATE OR REPLACE FUNCTION check_rate_limit(
  p_ip             text,
  p_max_requests   integer DEFAULT 10,
  p_window_seconds integer DEFAULT 60
)
RETURNS boolean
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_window  timestamptz;
  v_count   integer;
BEGIN
  -- Align to fixed window boundary
  v_window := to_timestamp(
    floor(extract(epoch FROM now()) / p_window_seconds) * p_window_seconds
  );

  INSERT INTO rate_limits (ip, window_start, count)
  VALUES (p_ip, v_window, 1)
  ON CONFLICT (ip, window_start)
  DO UPDATE SET count = rate_limits.count + 1
  RETURNING count INTO v_count;

  RETURN v_count <= p_max_requests;
END;
$$;

-- Grant execute to the anon/authenticated roles used by your supabase client
GRANT EXECUTE ON FUNCTION check_rate_limit(text, integer, integer)
  TO anon, authenticated;

-- ============================================================
-- Cleanup Function (call via pg_cron or manually)
-- ============================================================
CREATE OR REPLACE FUNCTION cleanup_rate_limits(
  older_than interval DEFAULT '1 year'
)
RETURNS void
LANGUAGE sql
SECURITY DEFINER
SET search_path = public
AS $$
  DELETE FROM rate_limits
  WHERE window_start < now() - older_than;
$$;

-- ============================================================
-- Schedule cleanup with pg_cron (enable via Supabase dashboard)
-- ============================================================
SELECT cron.schedule(
  'cleanup-rate-limits',
  '0 0 4 6- *',
  'SELECT cleanup_rate_limits();'
);



create or replace function sync_event_participants()
returns trigger as $$
declare
  target_event_id uuid;
begin
  if TG_OP = 'DELETE' then
    target_event_id := OLD.event_id;
  else
    target_event_id := NEW.event_id;
  end if;

  update events
  set participants = (
    select count(*) from participation where event_id = target_event_id
  )
  where id = target_event_id;

  return NEW;
end;
$$ language plpgsql;

create trigger trg_sync_participants
after insert or delete on participation
for each row execute function sync_event_participants();
